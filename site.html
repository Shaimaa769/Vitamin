<script>

document.getElementById('budget').addEventListener('input', function() {

document.getElementById('budgetValue').textContent = this.value;

});


document.getElementById('submitBtn').addEventListener('click', async function() {

const apiKey = document.getElementById('apiKey').value;

if (!apiKey) {

alert(' إدخال يرجى API Key');

return;

}


const age = document.getElementById('ageGroup').value;

const healthGoal = document.getElementById('healthGoal').value;

const medicalConditions = Array.from(

document.querySelectorAll('#medicalConditions input:checked')

).map(el => el.value);

const dietaryHabits = Array.from(

document.querySelectorAll('#dietaryHabits input:checked')

).map(el => el.value);

const allergies = document.getElementById('allergies').value;

const budget = document.getElementById('budget').value;


try {

const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {

method: 'POST',

headers: {

'Authorization': `Bearer ${apiKey}`,

'Content-Type': 'application/json'

},

body: JSON.stringify({

model: "llama-3.3-70b-versatile",

messages: [{

role: "user",

content: ` التالية المعايير على بناءً مناسبة غذائية مكملات 5-3 بتقديم قم :

{age}$ :العمر -

الصحي الهدف - : ${healthGoal}

الطبية الحالات - : ${medicalConditions.join(', ')}

الغذائية العادات - : ${dietaryHabits.join(', ')}

' || allergies}$ :الحساسية - يوجد لا '}

مصري جنيه {budget}$ :الميزانية -


التالي بالتنسيق البيانات تقديم يرجى :

الاسم .1

الرئيسية الفوائد .2

الاستخدام طريقة .3

التقديري السعر .4

)وجدت إن( إضافية تحذيرات .5


المصري السوق في المتوفرة المنتجات على التركيز مع .`

}],

max_tokens: 1000,

temperature: 0.7

})

});


const data = await response.json();

const recommendations = data.choices[0].message.content;


document.getElementById('supplementResults').innerHTML =

recommendations.split('\n').map(line =>

`<div class="supplement-card">${line}</div>`

).join('');


document.getElementById('resultsContainer').style.display = 'block';


} catch (error) {

console.error(' خطأ حدث :', error);

alert(' صحة من تأكد. التوصيات استرداد أثناء خطأ حدث API Key.');

}

});

</script>
